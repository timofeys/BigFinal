Процедура УдалитьДокументы() Экспорт
	Запрос = Новый Запрос("ВЫБРАТЬ
	                     |	Поступление.Ссылка
	                     |ИЗ
	                     |	Документ.Поступление КАК Поступление
	                     |ГДЕ
	                     |	Поступление.Дата > &Дата
	                     |
	                     |ОБЪЕДИНИТЬ ВСЕ
	                     |
	                     |ВЫБРАТЬ
	                     |	Продажа.Ссылка
	                     |ИЗ
	                     |	Документ.Продажа КАК Продажа
	                     |ГДЕ
	                     |	Продажа.Дата > &Дата");
	Запрос.УстановитьПараметр("Дата", НачалоДня(ТекущаяДата()));
	РезЗапроса = Запрос.Выполнить();
	Если НЕ РезЗапроса.Пустой() Тогда
		ДляУдаления = РезЗапроса.Выгрузить();
		Для Каждого стр из ДляУдаления Цикл
			ОбъектДляУдаления = стр.Ссылка.ПолучитьОбъект();
			ОбъектДляУдаления.Записать(РежимЗаписиДокумента.ОтменаПроведения);
			ОбъектДляУдаления.ПометкаУдаления = Истина;
			ОбъектДляУдаления.Записать();
		КонецЦикла;	
	КонецЕсли;	
КонецПроцедуры
